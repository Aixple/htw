\section{Ablauf}
	\subsection{Fragen zum Entwurf}
\begin{frame}
	    \note[item]<1>{Fragen des Entwerfenden die im Vorraus geklärt werden mussten}
	\begin{itemize}
		\item <2->Programmiersprache
		    \note[item]<2>{Programmiersprachen zur Auswahl: C/C++, Java, Python}
		    \note[item]<2>{Kriterium: GPIO ansteuerung}
		    \note[item]<2>{Einlesen einer XML oder Python Datei}
		\begin{itemize}
			\item <3->Python
			    \note[item]<3>{Trotz das nur 3 Leute richtig gut Python konnten}
		\end{itemize}
		\item <4->Dateiformat Konfigurationsdatei
		    \note[item]<4>{Einlesen einer Konfigurationsdatei}
		    \note[item]<4>{Am besten gleich in Variablen oder Objekt}
		    \note[item]<4>{Kleiner Test zeigte, geht in Python gut mit Json}
		\begin{itemize}
			\item <5->Json
		\end{itemize}
	\end{itemize}
\end{frame}

	\subsection{Erste Versuche}
\begin{frame}
	    \note[item]<1>{Erste Versuch unabhängig vom späteren Gesamtprodukt}
	    \note[item]<1>{Entscheidungshilfe für Programmiersprache und Dateiformat}
	\begin{itemize}
		\item <2->GPIO Pins ansteuern
		\item <3->Objekt in Json File speichern
		\item <4->Json File einlesen
	\end{itemize}
\end{frame}

	\subsection{Prototyp und erste Funktionen}
\begin{frame}
	    \note[item]<1>{Programm nimmt erste Formen an}
	\begin{itemize}
		\item <2->Funktionsköpfe des Entwurfs übernehmen
		    \note[item]<2>{Alle Funktionen die im Aktivitätsdiagramm umschriebene wurden}
		    \note[item]<2>{Nur Köpfe und ret definiert}
		\item <3->Erste Interaktion der Funktionen
		    \note[item]<3>{Datei als Argument übergeben und Infos auslesen}
		    \note[item]<3>{Status ändern und LEDs ansteuern}
		\item <4->Prototyp Funktionen: nicht fehlerfrei
		    \note[item]<4>{Funktionen waren noch nicht bereit für Tests}
		    \note[item]<4>{Probephase oder noch nicht erstellt}
	\end{itemize}
\end{frame}

	\subsection{Zuteilungen von Aufgaben}
\begin{frame}
	    \note[item]<1>{zunehmender Zeitdruck (auch andere Projekte)}
	\begin{itemize}
		\item <2->Qualitätsmanager: Datei-Funktionen
		    \note[item]<2>{einige Funktionen die sich mit der Dateiarbeit beschäftigen}
		    \note[item]<2>{Prüfen der Validität von Json Dateien}
		\item <3->Dokumentierer: Led-Funktionen
		    \note[item]<3>{einige Funktionen die sich mit den LEDs beschäftigen}
		    \note[item]<3>{Blink-Thread}
		\item <4->Datenbankbeauftragter: Datenbank-Funktionen
		    \note[item]<4>{Empfangen von Updates durch Sockets}
		\item <5->Kontrolle und Test
		    \note[item]<5>{Sichtprüfung und Funktionskontrolle der abgelieferten Funktionen}
		    \note[item]<5>{Einpflegen in den Prototyp}
	\end{itemize}
\end{frame}

	\subsection{Auslagerung der Funktionen}
\begin{frame}
	\begin{itemize}
		\item <2->Instruktion des Qualitätsmanagers
		    \note[item]<2>{Auf Anmerkungen des QM hin}
		
		\item <3->Dateien:
		\begin{itemize}
			\item <4->fileFunc.py
			        \note[item]<4>{Alle Funktionen welche sich mit der Erstellung oder Bearbeitung von Dateien beschäftigen.}
			\item <5->dbFunc.py
			        \note[item]<5>{Alle Funktionen welche sich mit der Datenbank beschäftigen.}
			\item <6->ledFunc.py
			        \note[item]<6>{Alle Funktionen welche sich mit dem Ansteuern der LEDs beschäftigen.}
			\item <7->stdFunc.py
			        \note[item]<7>{Alle Funktionen welche zur Entwicklung oder Fehlersuche des Programms nötig sind.}
			\item <8->ledleuchtturm.py
			        \note[item]<8>{Hauptdatei der Software, import ...Func.py Dateien und Aufruf der Funktionen}
		\end{itemize}
	\end{itemize}
\end{frame}

\end{frame}
